CREATE TABLE `KHOA` (
`MaKH` VARCHAR(2) NOT NULL ,
`TenKH` TEXT(50) NOT NULL ,
PRIMARY KEY (`MaKH`)
);


CREATE TABLE `MON` (
`MaMH` VARCHAR(2) NOT NULL ,
`TenMH` TEXT NOT NULL ,
`SoTiet` INT NOT NULL ,
PRIMARY KEY (`MaMH`)
);

CREATE TABLE `SINHVIEN` (
`MaSV` VARCHAR(3) NOT NULL ,
`HoSV` TEXT NOT NULL ,
`TenSV` TEXT NOT NULL ,
`Phai` BOOLEAN NOT NULL COMMENT 'Yes:Nu, No:Nam' ,
`NgaySinh` DATE NOT NULL COMMENT 'dd/mm/yy' ,
`NoiSinh` TEXT NOT NULL ,
`MaKH` VARCHAR(2) NOT NULL ,
`HocBong` INT NOT NULL ,
PRIMARY KEY(`MaSV`)
);

CREATE TABLE `KETQUA` (
`MaSV` VARCHAR(3) NOT NULL ,
`MaMH` VARCHAR(2) NOT NULL ,
`Diem` DOUBLE NOT NULL
);
