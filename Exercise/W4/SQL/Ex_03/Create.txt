CREATE DATABASE BT3;

CREATE TABLE Employee(
`Id` INT NOT NULL AUTO_INCREMENT,
`Name` VARCHAR(30) NOT NULL ,
`Address` VARCHAR(50) NOT NULL,
`Grade` VARCHAR(5) NOT NULL,
`Sex` VARCHAR(5) NOT NULL,
`Birthyear` int NOT NULL,
PRIMARY KEY(`Id`)
);

CREATE TABLE Grade(
`GradeID` VARCHAR(5) NOT NULL,
 `Salary` INT NOT NULL,
PRIMARY KEY(`GradeID`)
);
